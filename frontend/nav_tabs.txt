    <script>
        // Initialize API
        const api = new ProductivityAPI();

        // Load employee data
        async function loadEmployees() {
            try {
                const employees = await api.getEmployees();
                const tbody = document.getElementById('employeeTableBody');
                
                tbody.innerHTML = employees.map(emp => `
                    <tr>
                        <td>${emp.id}</td>
                        <td>${emp.name}</td>
                        <td>${emp.department || 'Unassigned'}</td>
                        <td>Team Member</td>
                        <td><span class="badge bg-success">Active</span></td>
                        <td>${emp.connecteam_user_id || 'N/A'}</td>
                        <td>
                            <button class="btn btn-sm btn-outline-primary" onclick="editEmployee(${emp.id})">
                                <i class="fas fa-edit"></i>
                            </button>
                        </td>
                    </tr>
                `).join('');
                
                document.getElementById('totalEmployees').textContent = employees.length;
            } catch (error) {
                console.error('Failed to load employees:', error);
            }
        }

        // Load active employees
        async function loadActiveEmployees() {
            try {
                const data = await api.getWorkingNow();
                document.getElementById('activeNow').textContent = data.active_count || 0;
            } catch (error) {
                console.error('Failed to load active employees:', error);
            }
        }

        // Sync functions
        async function syncConnecteam() {
            try {
                await api.syncEmployees();
                await api.syncTodayShifts();
                alert('Connecteam sync completed!');
                loadEmployees();
            } catch (error) {
                alert('Sync failed: ' + error.message);
            }
        }

        async function syncEmployees() {
            await syncConnecteam();
        }
